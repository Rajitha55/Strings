//Program to Reverse a given String

#include <bits/stdc++.h>
using namespace std;

string sreverse(string s)
{
    int start=0, end= s.length()-1;
    while(start<end)
    {
        swap(s[start], s[end]);
        start++;
        end--;
    }
    return s;
}

int main()
{
    string s;
    cout<<"Enter a string:"<<endl;
    cin>>s;
    //reverse(s.begin(), s.end()); -- Builtin function
    string rev=sreverse(s);
    cout<<rev;
    return 0;
}
