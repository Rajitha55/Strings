//Program to print number of vowels, consonants and white spaces

#include <bits/stdc++.h>
using namespace std;

void count(string s)
{
    int n=s.length();
    int vow=0, con=0, space=0;
    
    for(int i=0;i<n;i++)
    {
        if(s[i]=='a' || s[i]=='A' || s[i]=='e' || s[i]=='E' || s[i]=='i' || s[i]=='I' || s[i]=='o' || s[i]=='O' || s[i]=='u' || s[i]=='U')
           vow++;
        
        else if((s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z'))
           con++;
        
        else if(s[i]==' ')
           space++;
    }
    
    cout<<"No.of Vowels: "<<vow<<endl;
    cout<<"No.of Consonants: "<<con<<endl;
    cout<<"No.of White spaces: "<<space;
}

int main()
{
    string s;
    cout<<"Enter a string:";
    getline(cin,s);
    count(s);
    return 0;
}
