//Program to print count no.of words in a given string

#include <bits/stdc++.h>
using namespace std;

int word_count(string s)
{
    int len=s.length();
    int i=0,n=0;
    while(i<=len)
    {
        if(s[i]==' ' || s[i]=='\0')
           n++;
        i++;
    }
    return n;
}

int main()
{
    string s;
    cout<<"Enter string:"<<endl;
    getline(cin,s);
    int n=word_count(s);
    cout<<"No.of words present in given string are "<<n;
    return 0;
}
