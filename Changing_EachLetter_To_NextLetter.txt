//Program to change every letter in the given string with the letter following it in the alphabet (ie. a becomes b, p becomes q, z becomes a) 

#include <bits/stdc++.h>
using namespace std;

void change(string s)
{
    for(int i=0; i<s.length(); i++)
    {
        int ascii=(int)(s[i]);
        if(ascii == 122)
           s[i]='a';  //or s[i]=char(97)
        else if(ascii == 90)
           s[i]='A';  //or s[i]=char(65)
        else
           s[i]=char(ascii+1);
    }
    cout<<s;
}

int main()
{
    string s;
    cout<<"Enter String: ";
    cin>>s;
    change(s);
    return 0;
}
